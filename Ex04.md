variable_test.y
```

%{
#include <stdio.h>
#include <stdlib.h>

void yyerror(char *s);
int yytext(void);
%}

%token ID INT FLOAT DOUBLE

%%

S : T L ';'          { printf("\nValid declaration.\n"); }
  ;

L : L ',' ID         { printf("\nAnother variable: %s", yytext); }
  | ID               { printf("\nFirst variable: %s", yytext); }
  ;

T : INT              { printf("\nType: int"); }
  | FLOAT            { printf("\nType: float"); }
  | DOUBLE           { printf("\nType: double"); }
  ;

%%

int main() {
    printf("Enter declaration: ");
    yyparse();
    return 0;
}

void yyerror(char *s) {
    fprintf(stderr, "\nError: %s\n", s);
}


```
